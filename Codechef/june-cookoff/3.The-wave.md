# THE WAVE - MEDIUM LEVEL PROBLEM

### BRUTE FORCE APPROACH

```c++
#include <bits/stdc++.h>
#define ll long long int;
using namespace std;

// 3. The WAVE

string polynomial(int arr[], int n,int x){

    int cnt = 0;

    for(int i=0; i<n; i++){
        
        if(x-arr[i] == 0){
            return "0";
        }

        if(x-arr[i] <0 ){
            cnt++;
        }
    }

    string ans = (cnt%2==0)? "POSTITIVE" :"NEGATIVE";

    return ans;
}

int main() {

    int N,Q;
    cin>>N>>Q;

    int arr[N];

    for(int i=0; i<N; i++){
        cin>>arr[i];
    }

    // 1 3 5 100
    // p(x) = (x-1)(x-3)(x-5)(x-100)
    // -3 * -5* -7* -102
    // A (x-A)
    // Q1 =>

    // Brute force
    while(Q--){
        int x;
        cin>>x;
        
        cout<<polynomial(arr,N,x)<<endl;
      

    }

  

    return 0;
    
}
```
